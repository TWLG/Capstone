# HTTP: redirect everything to HTTPS
server {
    listen 80;
    server_name twlg.net;

    return 301 https://$host$request_uri;
}

# HTTPS: handle WebSocket traffic (no redirect here)
server {
    listen 443 ssl http2;
    server_name twlg.net;

    ssl_certificate /etc/letsencrypt/live/twlg.net/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/twlg.net/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

    # WebSocket endpoint
    location /ws/ {
        proxy_pass http://127.0.0.1:4000;
        proxy_http_version 1.1;

        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "Upgrade";

        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    location = /motor.html {
    	auth_basic "Restricted";
    	auth_basic_user_file /etc/nginx/.motor_passwd;

    	root /var/www/html;
    	try_files /motor.html =404;
    }

    # Optional: something harmless for normal HTTPS hits
    location / {
        return 204;
    }
}
